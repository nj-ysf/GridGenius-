# ── /etc/systemd/system/gridgenius-api.service ──────────────
[Unit]
Description=GridGenius FastAPI (port 8000)
After=network.target influxdb.service mosquitto.service
Wants=influxdb.service mosquitto.service

[Service]
User=pi
WorkingDirectory=/home/pi/GridGenius/ai
ExecStart=/usr/bin/python3 -m uvicorn api:app --host 0.0.0.0 --port 8000
Restart=always
RestartSec=5
Environment=PYTHONUNBUFFERED=1

[Install]
WantedBy=multi-user.target

# ── /etc/systemd/system/gridgenius-dashboard.service ─────────
[Unit]
Description=GridGenius Streamlit Dashboard (port 8501)
After=network.target gridgenius-api.service
Wants=gridgenius-api.service

[Service]
User=pi
WorkingDirectory=/home/pi/GridGenius/dashboard
ExecStart=/usr/bin/python3 -m streamlit run app.py --server.port 8501 --server.address 0.0.0.0 --server.headless true
Restart=always
RestartSec=5
Environment=PYTHONUNBUFFERED=1

[Install]
WantedBy=multi-user.target

# sudo cp gridgenius-api.service /etc/systemd/system/
